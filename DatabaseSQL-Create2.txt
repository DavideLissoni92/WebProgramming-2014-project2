CREATE TABLE "UTENTE" (
    "ID" INTEGER not null primary key GENERATED ALWAYS AS IDENTITY ( START WITH 1, INCREMENT BY 1),
    "USERNAME" VARCHAR(30) not null unique,
    "PASSWORD" VARCHAR(30),
    "AVATAR" VARCHAR(50) DEFAULT 'avatar/utente.jpg',
    "MODERATORE" BOOLEAN DEFAULT FALSE,
    "ULTIMOLOGIN" TIMESTAMP not null,
    "EMAIL" VARCHAR(100) not null,
    "RECUPEROPASSWORD" TIMESTAMP DEFAULT null

);

CREATE TABLE "GRUPPO" (
    "ID" INTEGER not null primary key GENERATED ALWAYS AS IDENTITY ( START WITH 1, INCREMENT BY 1),
    "NOME" VARCHAR(30) not null,
    "IDPROPR" INTEGER not null,
    "DATACREAZ" DATE not null,
    "PUBBLICO" BOOLEAN DEFAULT FALSE,
    "CHIUSO" BOOLEAN DEFAULT FALSE

);

CREATE TABLE "GRUPPO_UTENTE" (
    "ID" INTEGER not null primary key GENERATED ALWAYS AS IDENTITY ( START WITH 1, INCREMENT BY 1),
    "IDGRUPPO" INTEGER not null,
    "IDUTENTE" INTEGER not null,
	"ELIMINATO" BOOLEAN DEFAULT FALSE
);

CREATE TABLE "INVITO" (
    "ID" INTEGER not null primary key  GENERATED ALWAYS AS IDENTITY ( START WITH 1, INCREMENT BY 1),
    "IDGRUPPO" INTEGER not null,
    "IDUTENTE" INTEGER not null,
"ACCETTATO" BOOLEAN DEFAULT FALSE 
);

CREATE TABLE "POST" (
    "ID" INTEGER not null primary key  GENERATED ALWAYS AS IDENTITY ( START WITH 1, INCREMENT BY 1),
    "DATA" TIMESTAMP not null,
    "IDUTENTE" INTEGER not null,
    "IDGRUPPO" INTEGER not null ,
    "TESTO" VARCHAR(800) not null
);

CREATE TABLE "POST_FILE" (
    "ID" INTEGER not null primary key  GENERATED ALWAYS AS IDENTITY ( START WITH 1, INCREMENT BY 1),
    "IDPOST" INTEGER not null,
    "NOME" VARCHAR(30) not null
);
